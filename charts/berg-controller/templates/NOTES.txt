To create a ChallengeInstance:

  kubectl apply -f - <<EOF
  apiVersion: berg.norelect.ch/v1
  kind: ChallengeInstance
  metadata:
    name: example-instance
  spec:
    challengeRef:
      name: <challenge-name>
    ownerId: <uuid>
    flag: flag{example}
    timeout: 2h
  EOF
